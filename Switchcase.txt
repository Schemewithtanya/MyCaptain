import java.util.*;
class Main {
  public static void main(String[] args) {
    Scanner s=new Scanner(System.in);
    int i = s.nextInt();
    switch (i) {
      case 1:
        System.out.println("Case 1");
      case 2:
        System.out.println("Case 2");
      case 3:
        System.out.println("Case 3");

      default:
        System.out.println("Default case");
    }
  }
}
